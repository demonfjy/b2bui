“style”值为组件样式表名称，如页面有多个不同组件，此处可继承多个组件样式表。例如：页面有按钮组件，此处为"style" : ["page","button","..."]。“script”为js组件库，或其他js。“require”为此页面 js（自行编写）。“form1”与组件调用数组名称一致。

js调用
require(['show/show','select/select','checkbox/checkbox','radio/radio','file/file'], function() {
    var date_full=BL_Util.Date_full;
    var d=new date_full();
    var s_date=[];
    var e_date=[];
    $('#ticket-date').click(function(){ //方法是通过id绑定
    d.open({
            el:this,
            start:"",
        },function(v){
            e_date = v;
            var date=e_date[0]+"-"+e_date[1]+"-"+e_date[2];//年月日
            $('#ticket-date').val(date);//赋值
        });
    });
    $('#ticket-startdate').click(function(){ //方法是通过id绑定（开始时间）
        d.open({
            el:this,
            end:e_date,
            init:[2017,6],
        },function(v){
            s_date = v;
            var date=s_date[0]+"-"+s_date[1]+"-"+s_date[2];
            $('#ticket-startdate').val(date);
        });
    });
    $('#ticket-enddate').click(function(){//方法是通过id绑定（结束时间）
        d.open({
            el:this,
            start:s_date,
            init:[2017,6],
        },function(v){
            e_date = v;
            var date=e_date[0]+"-"+e_date[1]+"-"+e_date[2];
            $('#ticket-enddate').val(date);
        });
    });
});

此处调用的是组件效果的js，如果不调用组件js不会产生效果。
