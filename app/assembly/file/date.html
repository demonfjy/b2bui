时间控件 date

date 时间控件，组件有两种。
项目结构
{{#date1}} {{> date/date}} {{/date1}}

{{#date2}} {{> date/date}} {{/date2}} 
组件结构
{{#if sptype}}
    <span class="bl-calendar-picker">
        <span class="bl-input-affix-wrapper" style="width:{{width}}px;">
            <input readonly type="text" value="" id="{{date1.id}}" name="{{date1.name}}" class="bl-calendar-picker-input bl-input {{date1.class}}" {{#if date1.disabled}} disabled {{/if}} placeholder="{{date1.placeholder}}">
            <span class="bl-input-suffix"><i class="blicon blicon-calendar bl-input-search-icon"></i></span>
        </span>
    </span>
    <span class="datetxt">{{{sptext}}}</span>
    <span class="bl-calendar-picker">
        <span class="bl-input-affix-wrapper" style="width:{{width}}px;">
            <input readonly type="text" value="" id="{{date2.id}}" name="{{date2.name}}" class="bl-calendar-picker-input bl-input {{date2.class}}" {{#if date2.disabled}} disabled {{/if}} placeholder="{{date2.placeholder}}">
            <span class="bl-input-suffix"><i class="blicon blicon-calendar bl-input-search-icon"></i></span>
        </span>
    </span>
{{else}}
    <span class="bl-calendar-picker">
        <span class="bl-input-affix-wrapper">
            <input readonly type="text" value="" id="{{id}}" name="{{name}}" class="bl-calendar-picker-input bl-input {{class}}" {{#if disabled}} disabled {{/if}} placeholder="{{placeholder}}">
            <span class="bl-input-suffix"><i class="blicon blicon-calendar bl-input-search-icon"></i></span>
        </span>
    </span>

{{width}}控件宽度，{{{sptext}}}时间区间的连接符号可以是“-”、“到”、“至”，可自定义。

组件样式调用（sass文件或css文件）
@import "icon";  //icon
@import "tobe/components/date/date";  //日历控件样式
@import "tobe/components/input/input"; //日历输入框样式
@import "tobe/components/show/show";  //日历弹框样式

这是组件所需调用的组件样式。

组件结构调用
{{#date1}}
    {{> date/date}}
{{/date1}}
{{#date2}}
    {{> date/date}}
{{/date2}}

{{#date1}}、{{#date2}}为json数据名称，可自行定义。{{> date/date}}为组件在项目中的位置。

json数据结构
{
    "gl":{
        "title": "网页标题",
        "styles": ["css/date.css"],
        "scripts": ["http://k10.iblimg.com/resources/lib/jquery-1.8.2.js"],
        "require": "js/date.js"
    },
    "date1":{ 
        "id":"ticket-date", //id与input的id对应
        "name":"",
        "class":"bl-input-lg",
        "placeholder":"成立时间"
    },
    "date2":{
        "sptype":"1",
        "sptext":"至",          
        "style": "bl-col-xs-20 bl-col-sm-20 ",
        "width": "240",
        "date1":{
            "id":"ticket-startdate", 
            "name": "form-date", 
            "class" : " bl-input-lg",
            "placeholder": "开始时间"
        },
        "date2":{
            "id":"ticket-enddate", 
            "name": "form-date", 
            "class" : " bl-input-lg",
            "placeholder": "结束时间"
        } 
    }
}

“style”值为组件样式表名称，如页面有多个不同组件，此处可继承多个组件样式表。例如：页面有按钮组件，此处为"style" : ["page","button","..."]。“script”为js组件库，或其他js。“require”为此页面 js（自行编写）。“date1”、"date2"与组件调用数组名称一致。

js调用
require(['../../show/show'], function() {
    var date_full=BL_Util.Date_full;
    var d=new date_full();
    var s_date=[];
    var e_date=[];
    $('#ticket-date').click(function(){ //方法是通过id绑定
        d.open({
            el:this,
            start:"",
        },function(v){
            e_date = v;
            var date=e_date[0]+"-"+e_date[1]+"-"+e_date[2];//年月日
            $('#ticket-date').val(date);//赋值
        });
    });
    $('#ticket-startdate').click(function(){ //方法是通过id绑定（开始时间）
        d.open({
            el:this,
            end:e_date,
            init:[2017,6],
        },function(v){
            s_date = v;
            var date=s_date[0]+"-"+s_date[1]+"-"+s_date[2];
            $('#ticket-startdate').val(date);
        });
    });
    $('#ticket-enddate').click(function(){//方法是通过id绑定（结束时间）
        d.open({
            el:this,
            start:s_date,
            init:[2017,6],
        },function(v){
            e_date = v;
            var date=e_date[0]+"-"+e_date[1]+"-"+e_date[2];
            $('#ticket-enddate').val(date);
        });
    });
});

此处调用的是组件效果的js，如果不调用组件js不会产生效果。
